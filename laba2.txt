        static void Main(string[] args)

            (int, int, int, char) GetTuple(int[] array, string sentense)
            {
                int min = array[0];
                int max = array[0];
                int sum = 0;
                char letter = sentense[0];
                for (int i = 0; i < array.Length; i++)
                {
                    if (array[i] > min)
                    {
                        max = array[i];
                    }
                    if (array[i] < min)
                    {
                        min = array[i];
                    }
                    sum += array[i];
                }
                var tuple = (max, min, sum, letter);
                return tuple;
            }
